```python
"""Normalize free-text LLM path names to canonical career path labels.

Reads canonical paths, company-to-path overrides, and company name
aliases from search-config.json (generated by onboarding). Falls back
to pass-through if no config exists yet.
"""
from __future__ import annotations

import json
from pathlib import Path

CONFIG_PATH = Path(__file__).resolve().parents[2] / "data" / "search-config.json"

# Lazy-loaded cache
_cache: dict | None = None


def _load() -> dict:
    """Load and cache config. Returns empty structure if file missing."""
    global _cache
    if _cache is not None:
        return _cache

    _cache = {"paths": {}, "path_aliases": {}, "company_paths": {}, "company_aliases": {}}

    if not CONFIG_PATH.exists():
        return _cache

    try:
        config = json.loads(CONFIG_PATH.read_text(encoding="utf-8"))
    except (json.JSONDecodeError, OSError):
        return _cache

    # Build case-insensitive path lookup from query_pack labels
    for pack in config.get("query_packs", {}).values():
        label = pack.get("label", "")
        if label:
            _cache["paths"][label.strip().lower()] = label

    # Path aliases: fuzzy LLM variants -> canonical labels
    for variant, canonical in config.get("path_aliases", {}).items():
        _cache["path_aliases"][variant.strip().lower()] = canonical

    # Company-to-path overrides
    for company, path in config.get("company_path_overrides", {}).items():
        _cache["company_paths"][company.strip().lower()] = path

    # Company name aliases
    for variant, canonical in config.get("company_aliases", {}).items():
        _cache["company_aliases"][variant.strip().lower()] = canonical

    return _cache


def normalize_path(raw_name: str, canonical_paths: list[str] | None = None) -> str:
    """Map raw LLM path name to canonical label via case-insensitive lookup.

    Args:
        raw_name: The path name to normalize.
        canonical_paths: Ignored (kept for backwards compatibility).

    Returns:
        Canonical label if a match is found, otherwise raw_name unchanged.
    """
    if not raw_name:
        return raw_name
    cache = _load()
    key = raw_name.strip().lower()
    # Try exact match on query_pack labels first, then fuzzy aliases
    return cache["paths"].get(key) or cache["path_aliases"].get(key) or raw_name


def infer_path_for_company(company: str) -> str:
    """Return canonical path for a known company, or empty string."""
    if not company:
        return ""
    cache = _load()
    return cache["company_paths"].get(company.strip().lower(), "")


def normalize_company(name: str) -> str:
    """Normalize company name to canonical form via alias lookup."""
    if not name:
        return name
    cache = _load()
    return cache["company_aliases"].get(name.strip().lower(), name)


def get_canonical_paths() -> list[str]:
    """Return the list of canonical path labels from search-config.json."""
    cache = _load()
    return list(cache["paths"].values())
```

    path = tmp_path / "search-config.json"

class TestFindExisting:

Add these descriptions after the existing "Scoring keywords" section:

```
**Path aliases:** Map fuzzy LLM path variants to canonical query_pack labels. The LLM sometimes returns shortened or rephrased path names during evaluation. This map catches common variants so they normalize correctly. Generate 3-5 aliases per career path based on likely LLM rephrasings. Keys are lowercase variants, values are exact query_pack labels.

**Company path overrides:** Map specific company names to career paths. Used when a company should always be classified under a specific path regardless of how it was discovered. Keys are company names (case-insensitive at runtime), values are query_pack labels. Generate these from the user's target companies in each career path.

**Company aliases:** Map variant company names to their canonical form. Prevents duplicates when different sources use different names (e.g., "PwC Labs" -> "PwC", "Allvue" -> "Allvue Systems"). Generate initial aliases from the user's target companies that have known abbreviations or subsidiaries.
```

**Note:** These three fields are optional in `search_config_loader.py` (not in `REQUIRED_KEYS`). If missing, the normalizer falls back to pass-through behavior. This is intentional so existing configs keep working.
